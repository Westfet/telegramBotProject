# bat файл - файл сценария командной строки. Создание bat файла и запуск приложения через него -
хороший тон в разработке. В bat файле можно безопасно хранить что-то, например TOKEN

# эта команда нужна для того, чтобы служебная информация файла не "спамила" нам в консоль
@echo off

# команда активации виртуального окружения; %~dp0 - получение текущего расположения папки
call %~dp0Telegram_bot\venv\Scripts\activate

# переход командной строки в файл с нашим проектом
cd %~dp0Telegram_bot

# копируем сюда токен, чтобы безопасно его отсюда доставать
set TOKEN = 6054239011:AAHkkLUf_-eHbUog6E1DIETu41ZE4vllBA0

# запуск файла
python bot_telegram.py

# если у нас в скрипте возникнет ошибка, с этой командой окно не закроется, мы сможем увидеть ошибку
pause
